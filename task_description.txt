Практика работы с магическим методом __callStatic
Написать класс с реализацией метода __callStatic, который будет формировать SELECT запрос в зависимости от названия вызываемого метода и переданных аргументов.

findById($id); // SELECT * FROM <table> WHERE id = :id

findByEmailAndByStatus($email, $status); // SELECT * FROM <table> WHERE email = :email AND status = :status

findBetweenCreatedAt($startDate, $endDate); // SELECT * FROM <table> WHERE created_at BETWEEN :start_date AND :end_date

findBetweenCreatedAtAndByStatus($startDate, $endDate, $status); // SELECT * FROM <table> WHERE created_at BETWEEN :start_date AND :end_date AND status = :status

findBetweenCreatedAtAndInStatus($startDate, $endDate, $states); // SELECT * FROM <table> WHERE created_at BETWEEN :start_date AND :end_date AND status IN (:status_0, :status_1)
